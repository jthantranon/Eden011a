  7 ME3D.Builder = function (scene, physics) {
  8 	
  9 	// establish default values
 10 	var baseURL = 'models/base.json',
 11 		floorURL = 'models/floor.json',
 12 		crownURL = 'models/crown.json';
 13 		
 14 	
 15 		
 16 	
 17 	
 18 	// function globals
 19 	var buildingGroup = new THREE.Object3D;
 20 	buildingGroup.name = 'buildings';
 21 	scene.add(buildingGroup);
 22 		
 23 	var loader = new THREE.JSONLoader();
 24 	var material = new THREE.MeshBasicMaterial({color: 0x00FFFF, side:THREE.DoubleSide, opacity:.85, transparent:true});
 25 	var baseGeo, baseMesh;
 26 	var floorGeo, floorMesh;
 27 	var crownGeo, crownMesh;
 28 	var name = name || 'NoName';
 29 	
 30 	
 31 	
 32 	// GO!
 33 	this.makeBuilding = function(floors, scale, location, name)
 34 	{
 35 		var params = { floors:floors, scale:scale, location:location, name:name};
 36 		
 37 		var reference = loadBuilding(baseURL,params);
 38 		
 39 		return reference;
 40 	}
 41 		
 42 	
 43 	// HELPER FUNCTIONS
 44 	function loadBuilding(baseURL,params) {
 45 		loader.load(baseURL, function(loadedGeo) {
 46 			baseGeo = loadedGeo;
 47 			return loadFloor(floorURL,params); // load floor model next
 48 		});	
 49 	}
 50 	
 51 	function loadFloor(floorURL,params) {
 52 		loader.load(floorURL, function(loadedGeo) {
 53 			floorGeo = loadedGeo;
 54 			return loadCrown(crownURL,params); // load crown model next
 55 		});
 56 	}
 57 	
 58 	function loadCrown(crownURL,params) {
 59 		loader.load(crownURL, function(loadedGeo) {
 60 			crownGeo = loadedGeo;
 61 			return addBuilding(params); // now put it all together
 62 		});
 63 	}
 64 		
 65 	function addBuilding(params) {
 66 		
 67 		var location = params.location,
 68 			scale = params.scale,
 69 			floors = params.floors,
 70 			name = params.name;
 71 		
 72 		
 73 		// assemble meshes - geometry + material
 74 		var baseMesh =  new THREE.Mesh(baseGeo,  material);
 75 		var crownMesh = new THREE.Mesh(crownGeo, material);
 76 		
 77 		// establish initial positions
 78 		baseMesh.position.y += .25;
 79 		crownMesh.position.y += .75;
 80 		
 81 		// building container
 82 		var building = new THREE.Object3D();
 83 		building.name = name;
 84 		// add the base mesh since its ready			
 85 		building.add(baseMesh);
 86 		
 87 		// now add the floors and move the crown above them		
 88 		for(var i=1; i<=floors; i++) {
 89 			var floorMesh = new THREE.Mesh(floorGeo, material);
 90 			floorMesh.position.y += .25+i*.5;		
 91 			building.add(floorMesh);
 92 			crownMesh.position.y += .5;
 93 		}
 94 		
 95 		// add the crown
 96 		building.add(crownMesh);
 97 		
 98 		// position and scale the building
 99 		building.position = location;
100 		building.scale = scale;
101 		
102 		buildingGroup.add(building);
103 		
104 		console.log(building);
105 		
106 		makeBounds(building);
107 	}
108 	
109 	function makeBounds(building) {
110 		// get the building
111 		buildingMeshes = building.children;
112 		
113 		var boundingGeo = new THREE.Geometry();
114 				
115 		// get all the geometry, merge it
116 		for(var i=0,j=buildingMeshes.length; i<j; i++){
117 		  THREE.GeometryUtils.merge(boundingGeo,buildingMeshes[i]);
118 		};
119 		
120 		// compute bounding box for merged geometry
121 		boundingGeo.computeBoundingBox();
122 		var boundingSpec = boundingGeo.boundingBox.max;
123 		console.log(boundingSpec);
124 		
125 		physics.addMassBody(building,boundingSpec,0);
126 		
127 		// return new bounds vector3;
128 	}
129 	
130 	return this;
131 
132 };
133 
134 ME3D.Builder.prototype = {
135 	constructor: ME3D.Builder,
136 	
137 	log: function() {
138 		console.log(this);
139 	}
140 }
