  9 ME3D.Grid = function(unitSize, padding, totalUnits) {
 10 	
 11 	this.sizeRef = unitSize;
 12 	this.paddingRef = padding;
 13 	this.totalRef = totalUnits;
 14 	
 15 	var unitXPos = 0;
 16 	var unitYPos = .025;
 17 	var unitZPos = 0;
 18 	var opacity = .25;
 19 	this.gridObject = new THREE.Object3D();
 20 	
 21 	var gridNumbers = [];
 22 	var startNumber = Math.floor(-((totalUnits-1)/2));
 23 	
 24 	// make grid name array
 25 	for(var i=0,j=totalUnits; i<j; i++){
 26 		gridNumbers.push(startNumber);
 27 		startNumber++;	
 28 	};
 29 		
 30 	for(var i=0; i<totalUnits; i++) {
 31 		
 32 		for(var j=0; j<totalUnits; j++) {
 33 			var unitGeo = new THREE.PlaneGeometry(unitSize,unitSize);
 34 			var unitMat = new THREE.MeshBasicMaterial({color: 0x00FFFF, side:THREE.DoubleSide, opacity: opacity, transparent:true});
 35 			var unitMesh = new THREE.Mesh(unitGeo,unitMat);
 36 			
 37 			//unitMesh.name = gridLetters[i] + (gridNumbers[j]);
 38 			unitMesh.data = {codeName:"viper", coords:gridNumbers[j] + ',' + (gridNumbers[i])};
 39 			
 40 			unitMesh.useQuaternion = true;						
 41 			//unitMesh.rotation.x = ME3D.de2ra(-90);
 42 			var quaternion = new THREE.Quaternion();
 43 			quaternion.setFromAxisAngle( new THREE.Vector3( 1, 0, 0 ), ME3D.de2ra(-90) );
 44 			unitMesh.quaternion.copy(quaternion);
 45 			unitMesh.position.x = unitXPos;
 46 			unitMesh.position.y = unitYPos;
 47 			unitMesh.position.z = unitZPos;
 48 			this.gridObject.add(unitMesh);
 49 			
 50 			unitXPos += unitSize + padding;
 51 		}
 52 			
 53 			unitXPos = 0;
 54 			unitZPos += unitSize + padding;
 55 		
 56 	}
 57 	
 58 	//this.gridObject.position.x -= ((totalUnits*unitSize)/2)+(((totalUnits-1)/2)/2)-unitSize/2;
 59 	//this.gridObject.position.z -= ((totalUnits*unitSize)/2)+(((totalUnits-1)/2)/2)-unitSize/2;
 60 	//gridGroup = gridObject;
 61 	this.gridObject.name = 'grid';
 62 	
 63 	//return(gridObject);
 64 }
