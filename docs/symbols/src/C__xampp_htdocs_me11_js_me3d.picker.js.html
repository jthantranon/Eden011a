  7 ME3D.Picker = function (scene, camera) {
  8 	
  9 	var camera = camera;
 10 	var scene = scene;
 11 	
 12 	// MOUSEOVER PICKING!!! //
 13 	//////////////////////////
 14 	var mouse = { x: 0, y: 0 }, INTERSECTED, INTERSECTEDBUILDING;
 15 	
 16 	var isClicked=false;
 17 
 18 
 19 	
 20 	var projector = new THREE.Projector();
 21 	
 22 	document.addEventListener( 'mousemove', onDocumentMouseMove, false );
 23 	
 24 	
 25 	document.onmousedown = function(e) {
 26 		isClicked=true;	
 27 	}
 28 	
 29 	document.onmouseup = function(e) {
 30 		isClicked=false;
 31 	}
 32 	
 33 	
 34 	
 35 	function onDocumentMouseMove( event ) {
 36 	
 37 		event.preventDefault();
 38 	
 39 		mouse.x = ( event.clientX / window.innerWidth ) * 2 - 1;
 40 		mouse.y = - ( event.clientY / window.innerHeight ) * 2 + 1;
 41 	
 42 	}
 43 	
 44 	this.pick = function() {
 45 		// find intersections
 46 		
 47 		var vector = new THREE.Vector3( mouse.x, mouse.y, 1 );
 48 		projector.unprojectVector( vector, camera );
 49 	
 50 		var raycaster = new THREE.Raycaster( camera.position, vector.sub( camera.position ).normalize() );
 51 	
 52 		var scenegroups = scene.getChildByName('grid');
 53 		
 54 		var intersects = raycaster.intersectObjects( scenegroups.children );
 55 		
 56 		if ( intersects.length > 0 ) {
 57 			
 58 			
 59 			// if its not the same, clear the old one and update the new one		
 60 			if ( INTERSECTED != intersects[ 0 ].object ) {
 61 				if ( INTERSECTED ) INTERSECTED.material.opacity = .25;
 62 		
 63 				INTERSECTED = intersects[ 0 ].object;
 64 				INTERSECTED.material.opacity = .75;
 65 				if(isClicked) console.debug(INTERSECTED.data.coords);
 66 				isClicked=false;
 67 			// its the same or a new item
 68 			} else { 
 69 				INTERSECTED = intersects[ 0 ].object;
 70 				INTERSECTED.material.opacity = .75;
 71 				if(isClicked) console.debug(INTERSECTED.data.coords);
 72 				isClicked=false;
 73 			}
 74 		// nothing selected so set the currently lit one to normal
 75 		} else {
 76 		
 77 			if ( INTERSECTED ) INTERSECTED.material.opacity = .25;
 78 		
 79 			INTERSECTED = null;
 80 		
 81 		}
 82 	}
 83 	
 84 	
 85 	this.pickBuilding = function() {
 86 		// find intersections
 87 		
 88 		var vector = new THREE.Vector3( mouse.x, mouse.y, 1 );
 89 		projector.unprojectVector( vector, camera );
 90 	
 91 		var raycaster = new THREE.Raycaster( camera.position, vector.sub( camera.position ).normalize() );
 92 	
 93 		var scenegroups = scene.getChildByName('buildings');
 94 		//console.log(typeof scenegroups);
 95 		
 96 		if (typeof scenegroups == 'undefined') {
 97 			return false;
 98 		} 
 99 		
100 			var intersects = raycaster.intersectObjects( scenegroups.children, true);
101 			
102 			
103 			if ( intersects.length > 0 ) {
104 				
105 				
106 				// if its not the same, clear the old one and update the new one		
107 				if ( INTERSECTEDBUILDING != intersects[ 0 ].object ) {
108 					if ( INTERSECTEDBUILDING ) INTERSECTEDBUILDING.material.opacity = .85;
109 			
110 					INTERSECTEDBUILDING = intersects[ 0 ].object;
111 					INTERSECTEDBUILDING.material.opacity = 1;
112 					if(isClicked) console.debug('building');
113 					isClicked=false;
114 				// its the same or a new item
115 				} else { 
116 					INTERSECTEDBUILDING = intersects[ 0 ].object;
117 					INTERSECTEDBUILDING.material.opacity = 1;
118 					if(isClicked) console.debug('building');
119 					isClicked=false;
120 				}
121 			// nothing selected so set the currently lit one to normal
122 			} else {
123 			
124 				if ( INTERSECTEDBUILDING ) INTERSECTEDBUILDING.material.opacity = .85;
125 			
126 				INTERSECTEDBUILDING = null;
127 			
128 			}
129 		
130 	}
131 	
132 	return this;
133 };
134 
135 ME3D.Picker.prototype = {
136 	constructor: ME3D.Picker,
137 	
138 	log: function() {
139 		console.log(this);
140 	}
141 }
