  9 ME3D.Preloader = function() {
 10 	
 11 	this.onLoadComplete = {};
 12 	
 13 	this.loadList = [];
 14 	this.assets = {};
 15 	
 16 	this.listTotal;
 17 	this.listLoaded;
 18 	this.loadAttempts;
 19 	
 20 	this.jLoader = new THREE.JSONLoader();
 21 	this.tLoader = new THREE.TextureLoader();
 22 	this.sLoader = new THREE.SceneLoader();
 23 	
 24 	this.textureLoc = 'textures/';		
 25 };
 26 
 27 /*
 28  * API
 29  */
 30 
 31 ME3D.Preloader.prototype = {
 32 	constructor: ME3D.Preloader,
 33 	
 34 	log: function() {
 35 		ME3D.log(this);		
 36 	},
 37 	
 38 	add: function(url, handle) {
 39 		
 40 		scope = this;
 41 		
 42 		if(typeof url === 'undefined' || typeof handle === 'undefined') {
 43 			console.log('ME3D.Preloader: Error | Add - undefined arguments');
 44 			return false;
 45 		}
 46 		
 47 		for(var i=0,j=scope.loadList.length; i<j; i++){
 48 		  if(!scope.loadList[i][1].indexOf(handle)) {
 49 		  	console.log('ME3D.Preloader: Error | Add - handle in use: ' + handle);
 50 		  	console.log(scope.loadList);
 51 		  	console.log(scope.loadList[i][0]);
 52 		  	console.log(scope.loadList[i][1].indexOf(handle));
 53 		  	return false;
 54 		  };
 55 		};
 56 		
 57 		scope.loadList.push([url,handle]);
 58 		scope.listTotal = scope.loadList.length;
 59 		return true;
 60 				
 61 	},
 62 	
 63 	load: function(callback) {
 64 		
 65 		scope = this;
 66 		
 67 		if(typeof callback !== 'undefined') {
 68 			scope.onLoadComplete = callback;
 69 		}
 70 		
 71 		scope.listTotal = scope.loadList.length;
 72 		scope.loadedTotal = 0;
 73 		scope.loadAttempts = 9;
 74 		
 75 		for(var i=0,j=scope.loadList.length; i<j; i++){
 76 			
 77 			var url = scope.loadList[i][0];
 78 			var handle = scope.loadList[i][1];
 79 			
 80 			scope.jLoader.load(url, function(geometry, materials) {
 81 				scope.updateLoadProgress(geometry,materials,handle);
 82 			}, scope.textureLoc);
 83 		};
 84 					
 85 	},
 86 	
 87 	updateLoadProgress: function(geometry, materials, handle) {
 88 		
 89 		scope = this;
 90 		
 91 		console.log('loading ' + handle);
 92 		scope.assets[handle] = { geometry:geometry, materials:materials};
 93 		scope.loadedTotal++;
 94 		
 95 		if (scope.loadedTotal == scope.listTotal) {
 96 			console.log('preloading finished');
 97 			//console.log(scope.onLoadComplete);
 98 			scope.onLoadComplete();			
 99 		}
100 		
101 	}	
102 };
