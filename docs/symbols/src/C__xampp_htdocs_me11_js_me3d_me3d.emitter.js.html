  9 ME3D.Emitter = function () {
 10 	
 11 	this.age = 5;
 12 	this.amount = 6;
 13 	
 14 	// create the particle variables
 15 	var particleGroup = new THREE.Object3D();
 16 	particleGroup.name = 'particles';
 17 	
 18 	this.particleGeo = new THREE.Geometry();
 19     var particleMat = new THREE.ParticleBasicMaterial({
 20     	opacity: .25,
 21     	size: .25,
 22     	map: THREE.ImageUtils.loadTexture(
 23     		"textures/particle.png"),
 24     	blending: THREE.AdditiveBlending,
 25     	transparent: true, depthTest:false
 26    	});
 27    	
 28    	for(var i=0,j=this.amount; i<j; i++){
 29    		
 30    		var pX = Math.random() * .300 - .150,
 31    			pY = Math.random() * .300 - .150,
 32    			pZ = Math.random() * .300 - .150,
 33    			particle = new THREE.Vertex(
 34 	   			new THREE.Vector3(pX,pY,pZ));
 35    		//console.log(this.particleGeo.vertices);
 36    		this.particleGeo.vertices.push(particle);
 37    		
 38 	 };
 39 	 
 40 	 this.particleSystem = new THREE.ParticleSystem(
 41 	 	this.particleGeo, particleMat);
 42 	 	
 43 	 this.particleSystem.sortParticles = true;
 44 	 
 45 	 this.particleSystem.geometry.dirtyVertices = true;
 46 	 
 47 	 // update
 48 	 //////////////////////////////////
 49      this.update = function(delta) {
 50      	
 51      	var pCount = this.amount;
 52      	
 53      	while(pCount--) {
 54      		var particle = this.particleGeo.vertices[pCount];
 55      		if(particle.y > .75 ) {
 56      			particle.y = Math.random() * .300 - .150;
 57      		}
 58      		
 59      		var particleVelocity = Math.random() * .003;
 60      		var particleFade = Math.random() * .003;
 61      		particle.y += particleVelocity;
 62      		
 63      	}
 64      	
 65      	this.particleGeo.verticesNeedUpdate = true;
 66      	     	
 67      }
 68      
 69      // createParticle
 70 	 //////////////////////////////////     
 71      this.createParticle = function() {
 72      	
 73      }
 74      
 75      		
 76 };
 77 
 78 
 79 ME3D.Emitter.prototype = {
 80 	constructor: ME3D.Emitter,
 81 	
 82 	log: function() {
 83 		console.log(this);
 84 	},
 85 	
 86 	getSystem: function() {
 87 		return this.particleSystem;
 88 	}
 89 }
