 10 ME3D.Scene = function () {
 11 	
 12 	var self = this;
 13 	
 14 	// default scene properties
 15 	///////////////////////////////////
 16 	var width = $(window).width(), 
 17 		height = $(window).height(),
 18 		clock = new THREE.Clock();
 19 		
 20 		
 21 	var container = document.createElement( 'div' );
 22 	document.body.appendChild( container );	
 23 	
 24 	
 25 	// default camera attributes
 26 	///////////////////////////////////
 27 	var viewAngle = 60,
 28 		aspect = width/height,
 29 		near = 0.1,
 30 		far = 1000;
 31 	
 32 	
 33 	// default scene
 34 	///////////////////////////////////
 35 	self.scene = new THREE.Scene();
 36 	
 37 	
 38 	// default camera
 39 	///////////////////////////////////
 40 	self.camera = new THREE.PerspectiveCamera(
 41 		viewAngle, aspect, near, far);
 42 	self.camera.position.set(5,3,5);
 43 	self.camera.lookAt(new THREE.Vector3(0,0,0));
 44 				
 45 	self.scene.add(self.camera);
 46 	
 47 	
 48 	// default lights
 49 	///////////////////////////////////
 50 	self.sceneLight = new THREE.PointLight( 0xFFFFFF, 1, 10 );
 51 	self.sceneLight.position.set( 0, 3, 0);
 52 	self.scene.add( self.sceneLight );
 53 	
 54 	return this;
 55 	
 56 };
