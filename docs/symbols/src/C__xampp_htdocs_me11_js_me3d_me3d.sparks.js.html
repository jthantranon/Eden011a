 10 ME3D.Avatar = function () {
 11 	
 12 	// default properties
 13 	this.avatar = new THREE.Object3D();
 14 	this.body = new THREE.Mesh();
 15 	this.body.useQuaternion = true;
 16 	this.glowie = new THREE.ParticleSystem();
 17 	
 18 	
 19 	this.body.geometry = new THREE.SphereGeometry (.1,20,20);
 20 	this.body.material = new THREE.MeshBasicMaterial({color: 0x00FFFF, side:THREE.DoubleSide, opacity: .75, transparent:true});
 21 	this.avatar.position.x = 0;
 22 	this.avatar.position.y = 0;
 23 	this.avatar.position.z = 0;
 24 	
 25 	
 26 	this.cameraTarget = new THREE.Mesh(new THREE.CubeGeometry(.05,.05,.05), new THREE.MeshNormalMaterial());
 27 	this.cameraActual = new THREE.Mesh(new THREE.CubeGeometry(.025,.025,.025), new THREE.MeshNormalMaterial());
 28 	this.cameraTarget.position.set(2,2,2);
 29 	this.cameraActual.position.set(4,3,3);
 30 	this.cameraTarget.name = 'cameraTarget';
 31 	this.cameraActual.name = 'cameraActual';
 32 
 33 	
 34 	// create the particle variables
 35     var particles = new THREE.Geometry();
 36     var pMaterial = new THREE.ParticleBasicMaterial({
 37     	opacity: 1,
 38     	size: .75,
 39     	map: THREE.ImageUtils.loadTexture(
 40     		"textures/particle.png"),
 41     	blending: THREE.AdditiveBlending,
 42     	transparent: true,
 43    	});
 44    	
 45    	pMaterial.depthWrite = false;
 46 	var particle = this.body.position;
 47 	
 48 	particles.vertices.push(particle);
 49 	
 50 	// create the particle system
 51 	this.glowie.geometry = particles;
 52 	this.glowie.material = pMaterial;
 53 
 54 	this.glowie.sortParticles = true;
 55 	
 56 	//this.emitter
 57 	
 58 	this.avatar.add(this.body);
 59 	this.avatar.add(this.glowie);
 60 	this.avatar.add(this.cameraTarget);
 61 	this.avatar.add(this.cameraActual);
 62 	
 63 	this.avatar.dirtyVertices = true;
 64 	
 65 	// function getBoundsMesh() {
 66 		// return this.body;
 67 	// }	
 68 	
 69 	//return this.avatar;		
 70 
 71 };
 72 
 73 ME3D.Avatar.prototype = {
 74 	constructor: ME3D.Physics,
 75 	
 76 	getAvatar: function() {
 77 		return this.avatar;
 78 	},
 79 	
 80 	getBoundsMesh: function() {
 81 		return this.body;
 82 	},
 83 	
 84 	bindCamera: function(camera) {
 85 		
 86 		var newPosition = this.avatar.getChildPosition('cameraActual');
 87 		camera.position.set(newPosition);
 88 	},
 89 	
 90 	getCameraPosition: function(camera) {
 91 		
 92 		return this.avatar.getChildPosition('cameraActual');
 93 	}
 94 }
